seduce_success = {
	on_actions = {
			votc_seduce_succes
	}		
}

votc_seduce_succes = {
	effect = {
		if = {
			limit = {
				any_pinned_character = {
						has_character_flag = report_requested		
				}
			}
			add_character_flag = { flag = seduce_scheme_succes days = 180 }
			trigger_event = talk_event.9004
		}	
	}	
}
